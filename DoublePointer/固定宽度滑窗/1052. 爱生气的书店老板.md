public int maxSatisfied(int[] customers, int[] grumpy, int X) {

    int n = customers.length;
    int count = 0;

    for (int i = 0; i < n; ++i) {
        if (grumpy[i] == 0) {
            count += customers[i];
        }
    }

    for (int i = 0; i < X; ++i) {
        if (grumpy[i] == 1) {
            count += customers[i];
        }
    }

    int maxCount = count;
    for (int i = 0; i < n - X; ++i) {

        if (grumpy[i] == 1) {
            count -= customers[i];
        }

        if (grumpy[i + X] == 1) {
            count += customers[i + X];
        }

        maxCount = Math.max(maxCount, count);
    }

    return maxCount;
}
